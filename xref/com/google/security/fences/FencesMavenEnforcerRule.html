90    private void addFence(Fence f) throws MisconfigurationException {
91      f.check();
92      fences.add(f);
93    }

138   @Override
139   public void execute(EnforcerRuleHelper helper) throws EnforcerRuleException {
140     final Log log = helper.getLog();
141 
142     // TODO: maybe check MavenSession.getGoals() to see if this is being
143     // run at phase "validate" instead of phase "verify" to warn of a
144     // missing <phase>verify</phase> in the enforcer plugin configuration.
145 
146     ArtifactResolver resolver;
147     DependencyTreeBuilder treeBuilder;
148     ComponentConfigurator configurator;
149     try {
150       resolver = (ArtifactResolver) helper.getComponent(ArtifactResolver.class);
151       treeBuilder = (DependencyTreeBuilder)
152           helper.getComponent(DependencyTreeBuilder.class);
153 
154       // This seems "the right way" since plexus is supposed to inject
155       // dependencies, but when run without -X to turn on debugging,
156       // we get a MapOrientedComponentConfigurator which cannot configure
157       // this object.
158       // http://stackoverflow.com/questions/35919157/using-xmlplexusconfiguration-to-import-more-configuration-for-a-bean-style-maven
159       // explains the symptoms.
160       //  configurator = (ComponentConfigurator) helper.getComponent(
161       //      ComponentConfigurator.class);
162       configurator = new BasicComponentConfigurator();
163     } catch (ComponentLookupException ex) {
164       throw new EnforcerRuleException(
165           "Failed to locate component: " + ex.getLocalizedMessage(), ex);
166     }
167 
168     MavenProject project;
169     ArtifactRepository localRepository;
170     List<ArtifactRepository> remoteRepositories;
171     File buildDirectory;
172     try {
173       project = (MavenProject) helper.evaluate("${project}");
174       localRepository = (ArtifactRepository)
175           helper.evaluate("${localRepository}");
176       @SuppressWarnings("unchecked")
177       List<ArtifactRepository> rr = (List<ArtifactRepository>)
178           helper.evaluate("${project.remoteArtifactRepositories}");
179       remoteRepositories = rr;
180       buildDirectory = new File(
181           (String) helper.evaluate("${project.build.directory}"));
182     } catch (ExpressionEvaluationException ex) {
183       throw new EnforcerRuleException(
184           "Failed to locate component: " + ex.getLocalizedMessage(), ex);
185     }
186 
187     ArtifactFinder finder = new ArtifactFinder(
188         resolver, treeBuilder, localRepository, remoteRepositories, log);
189 
190     try {
191       finder.findClassRoots(project);
192     } catch (DependencyTreeBuilderException ex) {
193       throw new EnforcerRuleException("Failed to find artifacts", ex);
194     } catch (ArtifactResolutionException ex) {
195       throw new EnforcerRuleException("Failed to find artifacts", ex);
196     } catch (ArtifactNotFoundException ex) {
197       throw new EnforcerRuleException("Failed to find artifacts", ex);
198     } catch (MisconfigurationException ex) {
199       throw new EnforcerRuleException("Failed to find artifacts", ex);
200     }
201 
202     ImmutableList<ClassRoot> classRoots = finder.getClassRoots();
203 
204     InheritanceGraph inheritanceGraph;
205     try {
206       inheritanceGraph = InheritanceGraphExtractor
207           .fromClassRoots(classRoots);
208     } catch (IOException ex) {
209       throw new EnforcerRuleException(
210           "Failed to read classes to find inheritance relationships",
211           ex);
212     }
213 
214     try {
215       int nAssignedImportOrder = 0;
216       int importOrder = 0;
217       // Do the imports.
218       // Since an import might load a configuration that adds more imports, we
219       // just walk the list destructively.
220       for (; !imports.isEmpty(); ++importOrder) {
221         nAssignedImportOrder = rerootAndAssignImportOrder(
222             inheritanceGraph, nAssignedImportOrder, importOrder);
223         ConfigurationImport imp = imports.removeFirst();
224         if (alreadyImported.add(imp.key)) {
225           log.debug("Importing " + imp.key);
226           try {
227             imp.configure(
228                 this, configurator,
229                 new ConfigurationImport.ClassRoots(classRoots.iterator()),
230                 log);
231           } catch (MisconfigurationException ex) {
232             throw new EnforcerRuleException("Failed to import " + imp.key, ex);
233           }
234         } else {
235           log.info("Not importing " + imp.key + " a second time");
236         }
237       }
238       rerootAndAssignImportOrder(
239           inheritanceGraph, nAssignedImportOrder, importOrder);
240     } catch (MisconfigurationException ex) {
241       throw new EnforcerRuleException(ex.getMessage(), ex);
242     }
243 
244     ImmutableList<Fence> allFences = ImmutableList.copyOf(fences);
245 
246     if (allFences.isEmpty()) {
247       throw new EnforcerRuleException(
248           "No fences.  Please configure this rule with a policy."
249           + "  See https://github.com/mikesamuel/"
250           + "fences-maven-enforcer-rule/blob/master/src/site/markdown/usage.md"
251           + " for details");
252     }
253 
254     // Merge all the fences into one master.
255     final ApiFence mergedFence = new ApiFence();
256     for (Fence f : allFences) {
257       mergedFence.mergeDeep(f);
258     }
259 
260     // Log the effective configuration
261     boolean showConfig = RelevantSystemProperties.shouldShowEffectiveConfig();
262     if (showConfig || log.isDebugEnabled()) {
263       try {
264         Element config = mergedFence.buildEffectiveConfiguration();
265         TransformerFactory tf = TransformerFactory.newInstance();
266         Transformer transformer = tf.newTransformer();
267         transformer.setOutputProperty(OutputKeys.OMIT_XML_DECLARATION, "yes");
268         transformer.setOutputProperty(OutputKeys.METHOD, "xml");
269         transformer.setOutputProperty(OutputKeys.INDENT, "yes");
270         transformer.setOutputProperty(OutputKeys.ENCODING, "UTF-8");
271         transformer.setOutputProperty(
272             "{http://xml.apache.org/xslt}indent-amount", "2");
273 
274         StringWriter xmlOut = new StringWriter();
275         xmlOut.write("Effective Fences Rule Configuration:\n");
276         transformer.transform(
277             new DOMSource(config),
278             new StreamResult(xmlOut));
279         String xml = xmlOut.toString();
280         if (showConfig) {
281           log.info(xml);
282         } else {
283           log.debug(xml);
284         }
285       } catch (ParserConfigurationException ex) {
286         log.error(ex);
287       } catch (TransformerException ex) {
288         log.error(ex);
289       }
290     }
291 
292     File artifactFindingsFile = new File(buildDirectory, ".fences-cache.ser");
293     // TODO: store in the file, a hash of the effective policy so we can do
294     // a master check that the policy hasn't changed.
295     ArtifactFindingsHash afHash = ArtifactFindingsHash.readFrom(
296         Files.asByteSource(artifactFindingsFile));
297 
298     checkAllClasses(
299         project, log, inheritanceGraph, mergedFence, classRoots, afHash);
300 
301     ignore(artifactFindingsFile.getParentFile().mkdirs());
302     afHash.writeTo(Files.asByteSink(artifactFindingsFile));
303   }

305   private int rerootAndAssignImportOrder(
306       InheritanceGraph inheritanceGraph, int start, int importOrder)
307   throws MisconfigurationException {
308     int end = fences.size();
309     for (int i = start; i < end; ++i) {
310       Fence f = fences.get(i);
311       f = f.splitDottedNames(ApiElement.DEFAULT_PACKAGE, inheritanceGraph)
312           .promoteToApi();
313       f.assignImportOrder(importOrder);
314       fences.set(i, f);
315     }
316     return end;
317   }

319   protected static void checkAllClasses(
320       MavenProject project, Log backingLog, InheritanceGraph inheritanceGraph,
321       ApiFence mergedFence, Iterable<? extends ClassRoot> classRoots,
322       ArtifactFindingsHash afHash)
323   throws EnforcerRuleException {
324     final Policy p = Policy.fromFence(mergedFence);
325 
326     RecordingLog log = new RecordingLog(backingLog);
327 
328     log.debug(new LazyString() {
329       @Override
330       protected String makeString() {
331         return "Using policy\n" + p.toString();
332       }
333     });
334 
335     Checker checker = new Checker(log, inheritanceGraph, p);
336 
337     for (ClassRoot classRoot : classRoots) {
338       HashCode hashcode = null;
339 
340       Artifact art = classRoot.art;
341 
342       // If we have cached findings for this artifact, replay them instead
343       // of scanning the whole thing again.
344       String artId = art.getId();
345       switch (classRoot.kind) {
346         case ZIPFILE:
347           if (classRoot.classRoot.exists()) {
348             try {
349               hashcode = ArtifactFindingsHash.HASH_FUNCTION.hashBytes(
350                   Files.toByteArray(classRoot.classRoot));
351             } catch (IOException ex) {
352               log.warn("Trouble hashing zipfile", ex);
353             }
354           }
355           break;
356         default:
357           break;
358       }
359 
360       if (hashcode != null) {
361         Optional<ImmutableList<RecordingLog.Entry>> hashedResults =
362             afHash.get(artId, hashcode);
363         if (hashedResults.isPresent()) {
364           backingLog.debug("Replaying cached results for " + artId);
365           for (RecordingLog.Entry e : hashedResults.get()) {
366             e.apply(backingLog);
367           }
368           continue;
369         }
370       }
371 
372       log.info("Checking " + artId + " from scope " + art.getScope());
373       log.reset();
374       try {
375         checker.visitAll(ImmutableList.of(classRoot));
376       } catch (IOException ex) {
377         throw new EnforcerRuleException(
378             "Failed to check " + Utils.artToString(art), ex);
379       }
380       if (hashcode != null) {
381         afHash.store(artId, hashcode, log.getEntriesSinceLastReset());
382       }
383     }
384 
385     ImmutableList<Violation> violations = checker.getViolations();
386     PolicyViolationReporter reporter = new PolicyViolationReporter(log);
387     reporter.interpolator.addValueSource(
388         new PropertiesBasedValueSource(project.getProperties()));
389     int errorCount = reporter.report(violations);
390     if (errorCount != 0) {
391       String message = errorCount + " access policy violation"
392           + (errorCount == 1 ? "" : "s");
393       if (RelevantSystemProperties.inExperimentalMode()) {
394         log.info(message + " ignored in experimental mode");
395       } else {
396         throw new EnforcerRuleException(message);
397       }
398     }
399   }

418   static final class ArtifactFindingsHash implements Serializable {
419 
420     private static final long serialVersionUID = 3323962310197443927L;
421 
422     static final HashFunction HASH_FUNCTION = Hashing.sha512();
423 
424     private final Map<String, HashedResult> hashedResults =
425         Maps.newLinkedHashMap();
426 
427     Optional<ImmutableList<RecordingLog.Entry>> get(
428         String artId, HashCode hashcode) {
429       HashedResult r = hashedResults.get(artId);
430       if (r != null && r.getHashcode().equals(hashcode)) {
431         return Optional.of(r.getEntries());
432       }
433       return Optional.absent();
434     }
435 
436     void store(String artId, HashCode hashcode,
437                Iterable<? extends RecordingLog.Entry> entries) {
438       hashedResults.put(
439           artId,
440           new HashedResult(hashcode, ImmutableList.copyOf(entries)));
441     }
442 
443     static ArtifactFindingsHash readFrom(ByteSource byteSource)
444     throws EnforcerRuleException {
445       Object read;
446       try {
447         InputStream in = byteSource.openStream();
448         try {
449           ObjectInputStream objIn = new ObjectInputStream(in);
450           try {
451             read = objIn.readObject();
452             if (in.read() >= 0) {
453               throw new IOException("Extraneous content in " + byteSource);
454             }
455           } finally {
456             objIn.close();
457           }
458         } finally {
459           in.close();
460         }
461       } catch (ClassNotFoundException ex) {
462         throw new EnforcerRuleException(
463             "Failed to read artifact findings cache", ex);
464       } catch (@SuppressWarnings("unused") FileNotFoundException ex) {
465         // If there's no hash, just create a blank one.
466         read = new ArtifactFindingsHash();
467       } catch (IOException ex) {
468         throw new EnforcerRuleException(
469             "Failed to read artifact findings cache", ex);
470       }
471       return Preconditions.checkNotNull((ArtifactFindingsHash) read);
472     }
473 
474     void writeTo(ByteSink byteSink) throws EnforcerRuleException {
475       try {
476         OutputStream out = byteSink.openStream();
477         try {
478           ObjectOutputStream objOut = new ObjectOutputStream(out);
479           try {
480             objOut.writeObject(this);
481           } finally {
482             objOut.close();
483           }
484         } finally {
485           out.close();
486         }
487       } catch (IOException ex) {
488         throw new EnforcerRuleException(
489             "Failed to write artifact findings cache", ex);
490       }
491     }
492   }

494   static final class HashedResult implements Externalizable {
495     private HashCode hashcode;
496     private ImmutableList<RecordingLog.Entry> entries;
497 
498     public HashedResult() {
499       // for externalizable
500     }
501 
502     public ImmutableList<Entry> getEntries() {
503       return entries;
504     }
505 
506     public HashCode getHashcode() {
507       return hashcode;
508     }
509 
510     HashedResult(HashCode hashcode, ImmutableList<RecordingLog.Entry> entries) {
511       this.hashcode = hashcode;
512       this.entries = entries;
513     }
514 
515     @Override
516     public void writeExternal(ObjectOutput out) throws IOException {
517       out.writeObject(hashcode.asBytes());
518       out.writeObject(entries);
519     }
520 
521     @Override
522     public void readExternal(ObjectInput in)
523     throws IOException, ClassNotFoundException {
524       hashcode = HashCode.fromBytes((byte[]) in.readObject());
525       ImmutableList.Builder<RecordingLog.Entry> b = ImmutableList.builder();
526       for (Object entry : (Iterable<?>)  in.readObject()) {
527         b.add((RecordingLog.Entry) entry);
528       }
529       entries = b.build();
530     }
